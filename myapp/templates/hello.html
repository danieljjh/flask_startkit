这个 page 是从 pages/home 跳转过来的

提交 以下代码


    $('#submit').click(function () {
        var url = "http://127.0.0.1:5000/api/my_post";
        var d = JSON.stringify({
            name: "Yale",
            age: "14",
            education: {
                  school: "sxzz",
                  class: "m1"
              }
            });
        $.ajax({
            type: "POST",
            url: url,
            data: d,
            contentType: "application/json; charset=utf-8",
            dataType: "json",
            success: function (data) {
              $("#result").append(
                "name: " + data['name'] + " age: " + data['age'] + " school: " + data['school']
              )
            },
        })
    })
